Understanding the "path" 
               
                c.fillStyle="hotpink";
                c.strokeStyle="#000";
                c.lineWidth=2;
                
                c.rect(50, 50, 100, 200);
                c.fill();
                c.stroke();
                
                c.fillStyle="#7DF9FF";
                c.rect(250, 50, 100, 200);
                c.fill();
                c.stroke();
            
          
               
                c.beginPath();
                c.fillStyle="hotpink";
                c.strokeStyle="#000";
                c.lineWidth=2;
                c.rect(50, 50, 100, 200);
                c.fill();
                c.stroke();
                c.closePath();

Drawing custom shapes on the canvas
               
                c.beginPath();
                c.fillStyle="hotpink";
                c.strokeStyle="#000";
                c.lineWidth=2;
                c.moveTo(200, 50);
                c.lineTo(50, 200);
                c.lineTo(350, 200);
                c.lineTo(200, 50);
                c.fill();
                c.stroke();
                c.closePath();
